# 필요 패키지: playwright, playwright-stealth from playwright.async_api import async_playwright from playwright_stealth import Stealth async def scrape_nbc(): async with async_playwright() as p: browser = await p.chromium.launch(headless=False) # ← headless=True 는 차단됨 context = await browser.new_context( user_agent="Mozilla/5.0 ... Chrome/124.0.0.0 Safari/537.36", ) page = await context.new_page() await Stealth().apply_stealth_async(page) # Cloudflare 우회 # 1. 스마트폰 리뷰 목록 await page.goto("https://notebookcheck.net/Reviews.55.0.html") await page.wait_for_timeout(3000) links = await page.evaluate("""() => { return Array.from(document.querySelectorAll('a[href]')) .filter(a => /\\.\\d{6,}\\.0\\.html$/.test(a.href)) .map(a => ({title: a.innerText, href: a.href})); }""") # 2. 개별 리뷰 스크래핑 for link in links: await page.goto(link['href'], wait_until="domcontentloaded") await page.wait_for_timeout(2500) # CF 통과 대기 data = await page.evaluate("""() => ({ title: document.title, img: document.querySelector('meta[og:image]')?.content, desc: document.querySelector('meta[name=description]')?.content, paras: [...document.querySelectorAll('p')].map(p => p.innerText), pros: [...document.querySelectorAll('.nb-pros li')].map(l => l.innerText), cons: [...document.querySelectorAll('.nb-cons li')].map(l => l.innerText), })""")
